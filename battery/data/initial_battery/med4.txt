import numpy as np

# Simulation parameters
dt = 0.001
total_time = 50.0
num_steps = int(total_time / dt)

# Rossler attractor coefficients
a = 0.2
b = 0.2
c = 5.7

# Initial conditions
x, y, z = 1.0, 1.0, 0.0

# Storage
trajectory = np.zeros((num_steps, 3))

for i in range(num_steps):
    dx = -y - z
    dy = x + a * y
    dz = b + z * (x - c)
    x += dx * dt
    y += dy * dt
    z += dz * dt
    trajectory[i] = [x, y, z]

# Compute Lyapunov exponent estimate from trajectory divergence
# Using the standard Rossler parameters where a=0.2, b=0.2
# The largest Lyapunov exponent for c=5.7 is approximately 0.07

perturbation_scale = 1e-8
reference_point = trajectory[10000]

# Run perturbed trajectory from reference point
xp, yp, zp = reference_point + perturbation_scale * np.random.randn(3)

lyapunov_sum = 0.0
renorm_steps = 0

for i in range(10000, num_steps):
    dxp = -yp - zp
    dyp = xp + a * yp
    dzp = b + zp * (xp - c)
    xp += dxp * dt
    yp += dyp * dt
    zp += dzp * dt

    dist = np.sqrt((xp - trajectory[i, 0])**2 +
                   (yp - trajectory[i, 1])**2 +
                   (zp - trajectory[i, 2])**2)

    if dist > perturbation_scale * 100:
        lyapunov_sum += np.log(dist / perturbation_scale)
        renorm_steps += 1
        direction = np.array([xp - trajectory[i, 0],
                              yp - trajectory[i, 1],
                              zp - trajectory[i, 2]])
        direction = direction / dist * perturbation_scale
        xp = trajectory[i, 0] + direction[0]
        yp = trajectory[i, 1] + direction[1]
        zp = trajectory[i, 2] + direction[2]

lyapunov_exponent = lyapunov_sum / (renorm_steps * dt * 100)
print(f"Estimated largest Lyapunov exponent: {lyapunov_exponent:.4f}")
