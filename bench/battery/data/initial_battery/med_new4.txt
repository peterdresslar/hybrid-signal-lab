import sqlite3
from datetime import datetime

class TaskManager:
    def __init__(self, db_path="tasks.db"):
        self.conn = sqlite3.connect(db_path)
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS tasks (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                title TEXT NOT NULL,
                status TEXT DEFAULT 'pending',
                created_at TEXT,
                completed_at TEXT
            )
        """)
        self.conn.commit()

    def add_task(self, title):
        now = datetime.now().isoformat()
        self.cursor.execute(
            "INSERT INTO tasks (title, created_at) VALUES (?, ?)",
            (title, now)
        )
        self.conn.commit()
        return self.cursor.lastrowid

    def complete_task(self, task_id):
        now = datetime.now().isoformat()
        self.cursor.execute(
            "UPDATE tasks SET status='done', completed_at=? WHERE id=?",
            (now, task_id)
        )
        self.conn.commit()

    def get_pending(self):
        self.cursor.execute(
            "SELECT id, title FROM tasks WHERE status='pending'"
        )
        return self.cursor.fetchall()

    def close(self):
        self.conn.close()


if __name__ == "__main__":
    tm = TaskManager()
    task_id = tm.add_task("Write quarterly report")
    tm.add_task("Review pull requests")
    tm.add_task("Update documentation")

    pending = tm.get_pending()
    print(f"Pending tasks: {len(pending)}")

    tm.complete_task(task_id)

    pending = tm.get_pending()
    print(f"After completing one: {len(pending)}")

    tm.
